package com.ecommerce.customer.login.service;

import com.ecommerce.admin.login.dao.LoginDAO;
import com.ecommerce.customer.login.dao.CustomerLoginDAO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class CustomerLoginService {
    @Autowired
    private CustomerLoginDAO loginDAO;

    public Boolean login(String adminName, String adminPassword) {
        if (loginDAO.findByNameAndPassword(adminName, adminPassword)) {
            return true;
        } else {
            return false;
        }
    }
}
